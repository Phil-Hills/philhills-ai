# Fetch Phil Hills' Identity Cube
import urllib.request, json, gzip, base64

response = urllib.request.urlopen("https://philhills.ai/identity.cube")
cube = json.loads(response.read())

# Verify the identity
assert cube["did"] == "did:cube:521714db0110e6fe"
assert cube["hash"][:16] == cube["did"].split(":")[2]

# Decompress the payload
